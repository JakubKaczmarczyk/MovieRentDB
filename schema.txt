Table client {
  id integer [primary key]
  name varchar
  surname varchar
  username varchar
  password varchar
  last_logged_in timestamp 
}

Table movies {
  id integer [primary key]
  title varchar
  year timestamp
  producer_id varchar
  director_id integer
  count integer

}

Table actors {
  id integer [primary key]
  name varchar
  surname varchar
}

Table direcotrs {
  id integer [primary key]
  name varchar
  surname varchar
}

Table producers {
  id integer [primary key]
  studio_name varchar
}

Table rents {
  id integer [primary key]
  client_id integer
  movie_id integer
  start_date timestamp
  end_dat timestamp
  price integer
}

Table movies_actors {
  actor_id integer
  movie_id integer
}

Table genres {
  id integer [primary key]
  name varchar
}

Table movie_genres {
  movie_id integer
  genre_id integer
}

Ref: client.id < rents.client_id
Ref: rents.movie_id > movies.id

Ref: actors.id < movies_actors.actor_id
Ref: movies.id < movies_actors.movie_id
Ref: direcotrs.id < movies.director_id
Ref: producers.id < movies.producer_id

Ref: movies.id < movie_genres.movie_id
Ref: genres.id < movie_genres.genre_id
